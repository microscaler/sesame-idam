```rust
use crate::key_manager::KEY_MANAGER;
use crate::key_manager::KeyError;

#[handler(AdminRevokeKeyController)]
pub fn handle(req: TypedHandlerRequest<Request>) -> Response {
    let kid = req.data.kid.clone();
    let span = tracing::span!(tracing::Level::INFO, "key.revoke.admin", kid = &kid);
    let _guard = span.enter();

    let result = KEY_MANAGER
        .write()
        .map_err(|e| {
            tracing::error!(error = ?e, "KEY_MANAGER write lock poisoned");
        })
        .and_then(|mut km| km.revoke_key(&kid).map_err(|e| (kid.clone(), e)));

    match result {
        Ok(_) => {
            tracing::info!(kid = &kid, "admin: key revoked successfully");
            span.record("result", "success");
            Response {
                kid: Some(kid),
                message: Some("Key revoked successfully".to_string()),
                success: Some(true),
            }
        }
        Err((kid, KeyError::KeyNotFound)) => {
            tracing::warn!(kid = &kid, "admin: revocation requested for unknown key");
            span.record("result", "key_not_found");
            Response {
                kid: Some(kid),
                message: Some("Key not found".to_string()),
                success: Some(false),
            }
        }
        Err((kid, KeyError::NoKeyToRevoke)) => {
            tracing::warn!(kid = &kid, "admin: key is not currently active or pending");
            span.record("result", "no_key_to_revoke");
            Response {
                kid: Some(kid),
                message: Some("Key is not currently active".to_string()),
                success: Some(false),
            }
        }
        Err((kid, e)) => {
            tracing::error!(kid = &kid, error = ?e, "admin: revocation failed unexpectedly");
            span.record("result", "error");
            Response {
                kid: Some(kid),
                message: Some(format!("Revocation failed: {e}")),
                success: Some(false),
            }
        }
    }
}
```
